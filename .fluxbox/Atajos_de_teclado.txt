<--Atajos de teclado de fluxbox-->

<--cambiar de ventana-->
   win + tab

<--cambiar de escritorio-->
   win + 0-9

<--mover una ventana de un estritorio a otro-->
   win + shift + 0-9

<--mover una ventana y cambiar de escritorio-->
   win + ctrl + 0-9

<--atajos de ventana-->>
   win + shift + q				(cerrar)
   win + F9						(minimizar)
   win + F10					(maximizar)
   win + F11					(pantalla completa)
   alt + click derecho			(redimencionar)
   alt + click izquierdo		(mover)

<--salir de fluxbox-->
   win + shift + F11

<--abrir programas-->
   win + return (lxterminal)
   win + d      (dmenu)
   win + w      (firefox)
   win + e      (thunar)
